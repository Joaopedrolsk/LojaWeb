CREATE DATABASE IF NOT EXISTS Loja_DB;
CREATE user 'loja'@'localhost' identified by 'loja';
grant all privileges on . to 'loja'@'localhost';
select * from mysql.user;


flush privileges;


use loja_db;


-- ===========================
-- Tabela: CLIENTES
-- ===========================
CREATE TABLE CLIENTES (
    id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cpf CHAR(11) NOT NULL UNIQUE,
    email VARCHAR(100),
    telefone VARCHAR(15),
    data_nascimento DATE,
    ativo BOOLEAN DEFAULT TRUE
) ENGINE=InnoDB;


-- ===========================
-- Tabela: ENDERECOS
-- ===========================
CREATE TABLE ENDERECOS (
    id_endereco INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT NOT NULL,
    logradouro VARCHAR(150),
    numero VARCHAR(10),
    complemento VARCHAR(100),
    bairro VARCHAR(60),
    cidade VARCHAR(60),
    estado CHAR(2),
    cep CHAR(8),
    CONSTRAINT fk_enderecos_clientes
        FOREIGN KEY (id_cliente) REFERENCES CLIENTES(id_cliente)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    INDEX idx_cliente_endereco (id_cliente)
) ENGINE=InnoDB;


-- ===========================
-- Tabela: CATEGORIAS
-- ===========================
CREATE TABLE CATEGORIAS (
    id_categoria INT AUTO_INCREMENT PRIMARY KEY,
    nome_categoria VARCHAR(100) NOT NULL,
    descricao TEXT
) ENGINE=InnoDB;


-- ===========================
-- Tabela: PRODUTOS
-- ===========================
CREATE TABLE PRODUTOS (
    id_produto INT AUTO_INCREMENT PRIMARY KEY,
    nome_produto VARCHAR(100) NOT NULL,
    preco DECIMAL(10,2) NOT NULL,
    estoque INT DEFAULT 0,
    id_categoria INT,
    CONSTRAINT fk_produtos_categorias
        FOREIGN KEY (id_categoria) REFERENCES CATEGORIAS(id_categoria)
        ON DELETE SET NULL
        ON UPDATE CASCADE,
    INDEX idx_categoria_produto (id_categoria)
) ENGINE=InnoDB;


-- ===========================
-- Tabela: FUNCIONARIOS
-- ===========================
CREATE TABLE FUNCIONARIOS (
    id_funcionario INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cargo VARCHAR(50),
    salario DECIMAL(10,2),
    data_admissao DATE
) ENGINE=InnoDB;


-- ===========================
-- Tabela: PEDIDOS
-- ===========================
CREATE TABLE PEDIDOS (
    id_pedido INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT NOT NULL,
    id_funcionario INT,
    data_pedido DATETIME DEFAULT CURRENT_TIMESTAMP,
    status ENUM('aberto', 'processando', 'enviado', 'entregue', 'cancelado') DEFAULT 'aberto',
    CONSTRAINT fk_pedidos_clientes
        FOREIGN KEY (id_cliente) REFERENCES CLIENTES(id_cliente)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    CONSTRAINT fk_pedidos_funcionarios
        FOREIGN KEY (id_funcionario) REFERENCES FUNCIONARIOS(id_funcionario)
        ON DELETE SET NULL
        ON UPDATE CASCADE,
    INDEX idx_cliente_pedido (id_cliente),
    INDEX idx_funcionario_pedido (id_funcionario)
) ENGINE=InnoDB;


-- ===========================
-- Tabela: PAGAMENTOS
-- ===========================
CREATE TABLE PAGAMENTOS (
    id_pagamento INT AUTO_INCREMENT PRIMARY KEY,
    id_pedido INT NOT NULL,
    id_funcionario INT,
    valor DECIMAL(10,2) NOT NULL,
    data_pagamento DATETIME DEFAULT CURRENT_TIMESTAMP,
    metodo ENUM('cartao', 'boleto', 'pix', 'dinheiro') NOT NULL,
    CONSTRAINT fk_pagamentos_pedidos
        FOREIGN KEY (id_pedido) REFERENCES PEDIDOS(id_pedido)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    CONSTRAINT fk_pagamentos_funcionarios
        FOREIGN KEY (id_funcionario) REFERENCES FUNCIONARIOS(id_funcionario)
        ON DELETE SET NULL
        ON UPDATE CASCADE,
    INDEX idx_pedido_pagamento (id_pedido),
    INDEX idx_funcionario_pagamento (id_funcionario)
) ENGINE=InnoDB;


-- ===========================
-- Tabela: AUDITORIA
-- ===========================
CREATE TABLE AUDITORIA (
    id_auditoria INT AUTO_INCREMENT PRIMARY KEY,
    id_funcionario INT,
    acao VARCHAR(100) NOT NULL,
    tabela_afetada VARCHAR(100),
    data_hora DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuario VARCHAR(100),
    CONSTRAINT fk_auditoria_funcionarios
        FOREIGN KEY (id_funcionario) REFERENCES FUNCIONARIOS(id_funcionario)
        ON DELETE SET NULL
        ON UPDATE CASCADE,
    INDEX idx_funcionario_auditoria (id_funcionario)
) ENGINE=INNODB;
